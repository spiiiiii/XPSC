#include <iostream>
 
using namespace std;
 
int main()
{
    int t;cin>>t;while(t--){
        int n,y=0,x=0;cin>>n;
        string s;
        cin>>s;
        bool pos =false;
        for(int i = 0 ; i < n ; i++){
            if(s[i]=='U'){
               y++;
            }
             if(s[i]=='D'){
                y--;
            }
             if(s[i]=='L'){
                x--;
            }
             if(s[i]=='R'){
                x++;
            }
            if(x==1 && y==1){
                pos =true;
                break;
            }
        }
        
        if(pos==true) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
        
    }
    
 
    return 0;
}
