#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t; 

    while (t--) {
       string d;
        cin >> d; 

        int c = 0;
        string a = "314159265358979323846264338327";

        for (int i = 0; i < d.length(); i++) {
            if (i >= a.length() || d[i] != a[i]) {
                break;
            }
            c++;
        }

        cout << c << endl; 
    }

    return 0;
}
